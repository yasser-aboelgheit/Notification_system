24 """ 

25 !!!! IMPORTANT !!!! MUST raise a TokenError with a user-facing error 

26 message if the given token is invalid, expired, or otherwise not safe 

27 to use. 

28 """ 

29 if self.token_type is None or self.lifetime is None: 

30 raise TokenError(_('Cannot create token with no type or lifetime')) 

31 

32 self.token = token

84 def verify(self): 

85 """ 

86 Performs additional validation steps which were not performed when this 

87 token was decoded. This method is part of the "public" API to indicate 

88 the intention that it may be overridden in subclasses. 

89 """ 

90 # According to RFC 7519, the "exp" claim is OPTIONAL 

91 # (https://tools.ietf.org/html/rfc7519#section-4.1.4). As a more 

92 # correct behavior for authorization tokens, we require an "exp" 

93 # claim. We don't want any zombie tokens walking around. 

94 self.check_exp() 

95 

96 # Ensure token id is present 

97 if api_settings.JTI_CLAIM not in self.payload: 

98 raise TokenError(_('Token has no id')) 

99 

100 self.verify_token_type()

102 def verify_token_type(self): 

103 """ 

104 Ensures that the token type claim is present and has the correct value. 

105 """ 

106 try: 

107 token_type = self.payload[api_settings.TOKEN_TYPE_CLAIM] 

108 except KeyError: 

109 raise TokenError(_('Token has no type')) 

110 

111 if self.token_type != token_type: 

112 raise TokenError(_('Token has wrong type'))

114 def set_jti(self): 

115 """ 

116 Populates the configured jti claim of a token with a string where there 

117 is a negligible probability that the same string will be chosen at a 

118 later time. 

119 

120 See here: 

121 https://tools.ietf.org/html/rfc7519#section-4.1.7 

122 """ 

123 self.payload[api_settings.JTI_CLAIM] = uuid4().hex 

124 

125 def set_exp(self, claim='exp', from_time=None, lifetime=None): 

126 """ 

127 Updates the expiration time of a token. 

128 """ 

129 if from_time is None: 

130 from_time = self.current_time 

131 

132 if lifetime is None: 

133 lifetime = self.lifetime 

134 

135 self.payload[claim] = datetime_to_epoch(from_time + lifetime) 

136 

137 def check_exp(self, claim='exp', current_time=None): 

138 """ 

139 Checks whether a timestamp value in the given claim has passed (since 

140 the given datetime value in `current_time`). Raises a TokenError with 

141 a user-facing error message if so. 

142 """ 

143 if current_time is None: 

144 current_time = self.current_time 

145 

146 try: 

147 claim_value = self.payload[claim] 

148 except KeyError: 

149 raise TokenError(format_lazy(_("Token has no '{}' claim"), claim)) 

150 

151 claim_time = datetime_from_epoch(claim_value) 

152 if claim_time <= current_time: 

153 raise TokenError(format_lazy(_("Token '{}' claim has expired"), claim))

184 def check_blacklist(self): 

185 """ 

186 Checks if this token is present in the token blacklist. Raises 

187 `TokenError` if so. 

188 """ 

189 jti = self.payload[api_settings.JTI_CLAIM] 

190 

191 if BlacklistedToken.objects.filter(token__jti=jti).exists(): 

192 raise TokenError(_('Token is blacklisted'))

194 def blacklist(self): 

195 """ 

196 Ensures this token is included in the outstanding token list and 

197 adds it to the blacklist. 

198 """ 

199 jti = self.payload[api_settings.JTI_CLAIM] 

200 exp = self.payload['exp'] 

201 

202 # Ensure outstanding token exists with given jti 

203 token, _ = OutstandingToken.objects.get_or_create( 

204 jti=jti, 

205 defaults={ 

206 'token': str(self), 

207 'expires_at': datetime_from_epoch(exp), 

208 }, 

209 ) 

210 

211 return BlacklistedToken.objects.get_or_create(token=token)

265 @property 

266 def access_token(self): 

267 """ 

268 Returns an access token created from this refresh token. Copies all 

269 claims present in this refresh token to the new access token except 

270 those claims listed in the `no_copy_claims` attribute. 

271 """ 

272 access = AccessToken() 

273 

274 # Use instantiation time of refresh token as relative timestamp for 

275 # access token "exp" claim. This ensures that both a refresh and 

276 # access token expire relative to the same time if they are created as 

277 # a pair. 

278 access.set_exp(from_time=self.current_time) 

279 

280 no_copy = self.no_copy_claims 

281 for claim, value in self.payload.items(): 

282 if claim in no_copy: 

283 continue 

284 access[claim] = value 

285 

286 return access
